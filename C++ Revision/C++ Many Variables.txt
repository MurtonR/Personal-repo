#include <iostream>
using namespace std;

int main() {
  int x = 5, y = 6, z = 50, a = 90;  -Multiple variables of the same type can be declared by using a comma between each one
  cout << x + y + z + a;
  return 0;
}



#include <iostream>
using namespace std;

int main() {
  int x, y, z;
  x = y = z = 50;      -Multiple of variables of the same type can be declared and assigned the same value in one line
  cout << x + y + z;  
  return 0;
}
